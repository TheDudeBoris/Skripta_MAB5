\begin{enumerate}%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\item \VETA  o zachov\'an\'i vlastnost\'i hustoty pravd\v epodobnosti
\item \VETA $f,g$ jsou hustoty pravd\v epodobnosti a $\int_\R xf(x)~\dx=\mu_1$ a $\int_\R yg(y)~\dy=\mu_2,$ pak $\int_\R z(f\star g)(z)~\dz=\mu_1+\mu_2$
\item \VETA  o posunut\'i v konvoluci v $\LLL_1(\Er)$
\item \VETA  o derivaci v konvoluci:  $f(\x)\in\LLL_1(\Er)$ a $g(\x)\in\LLL_1(\Er)\cap \CC_0^1$
\item ortonormální množina: \v Rekneme, \v ze mno\v zina $S$ z Hilbertova prostoru $\H$ je ortonorm\' aln\' i, pokud pro ka\v zdou funkci $f(\x)\in S$ je $\|f(\x)\|=1$ a z\' arove\v n pro ka\v zd\'e dv\v e funkce $f(\x),g(\x)\in S$ takov\'e, \v ze $f(\x)\neq g(\x)$ plat\' i rovnost $\bla f|g\bra=0.$
\item \LEMMA prvky ka\v zd\'e ortonormální množiny jsou LN
\item zmínka o Grammovì-Schmidtovì proceduøe vyrábìjící ON množinu z množiny LN funkcí
\item \VETA  o Besselovì nerovnosti: Nech\v t $S=\{f_1(\x),f_2(\x),\ldots,f_n(\x)\}$ je (kone\v cn\'a) ortonorm\'aln\'i mno\v zina v Hilbertov\v e prostoru $\H.$ Nech\v t $g(\x)\in\H$ je zvolen libovoln\v e. Ozna\v cme $a_k:=\la g|f_k\ra.$ Pak plat\'i
%
\BE \sum_{k=1}^n |a_k|^2 \leq \|g(\x)\|^2. \label{Bessel-ineq}\EE
